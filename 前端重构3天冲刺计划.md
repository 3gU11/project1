router.beforeEach(async (to, from, next) => {
  const authStore = useAuthStore()
  
  if (to.meta.requiresAuth && !authStore.token) {
    next('/login')
    return
  }
  
  if (to.meta.permissions && !hasPermission(to.meta.permissions)) {
    next('/403')
    return
  }
  
  next()
})
